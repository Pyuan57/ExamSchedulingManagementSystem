CREATE DATABASE ExamSchedulingManagementSystem;

CREATE TABLE FACULTY(
Fac_Code VARCHAR(4) NOT NULL,
Fac_Name VARCHAR(50),
PRIMARY KEY(Fac_Code));

CREATE TABLE LECTURER(
Lec_ID CHAR(10) NOT NULL,
Lec_Name VARCHAR(50),
Lec_Address VARCHAR(150),
Lec_Ctnum VARCHAR(12),
Fac_Code VARCHAR(4) NOT NULL,
Lec_Email VARCHAR(30),
PRIMARY KEY(Lec_ID),
FOREIGN KEY (Fac_Code) REFERENCES FACULTY(Fac_Code)
ON DELETE RESTRICT ON UPDATE CASCADE);

CREATE TABLE STUDENT(
Std_ID CHAR(10) NOT NULL,
Std_Name VARCHAR(50),
Std_Birth DATE,
Std_IC VARCHAR(14),
Std_Ctnum VARCHAR(12),
Std_Address VARCHAR(150),
Fac_Code VARCHAR(4) NOT NULL,
Std_Nation VARCHAR(20),
Std_Grade DECIMAL(4,2),
Std_InMonth VARCHAR(2),
Std_Email VARCHAR(45),
PRIMARY KEY(Std_ID),
FOREIGN KEY (Fac_Code) REFERENCES FACULTY(Fac_Code)
ON DELETE RESTRICT ON UPDATE CASCADE);

CREATE TABLE SUBJECT (
Sub_Code VARCHAR(9) NOT NULL,
Sub_Name VARCHAR(45),
Fac_Code VARCHAR(4) NOT NULL,
Credit_Hour INTEGER(1),
Ex_Time DECIMAL(2,1) NOT NULL,
PRIMARY KEY (Sub_Code),
FOREIGN KEY (Fac_Code) REFERENCES FACULTY(Fac_Code)
ON DELETE RESTRICT ON UPDATE CASCADE);

CREATE TABLE VENUE (
Venue_Code VARCHAR(8) NOT NULL,
Venue_Name VARCHAR(30),
Venue_Capacity INTEGER(3) NOT NULL,
PRIMARY KEY (Venue_Code));

CREATE TABLE BOOKING (
Book_Code VARCHAR(6) NOT NULL,
Book_Date DATE NOT NULL,
Book_Stime TIME NOT NULL,
Book_Etime TIME NOT NULL,
Venue_Code VARCHAR(8) NOT NULL,
Sub_Code VARCHAR(9) NOT NULL,
PRIMARY KEY (Book_Code),
FOREIGN KEY (Venue_Code) REFERENCES VENUE(Venue_Code)
ON DELETE RESTRICT ON UPDATE CASCADE,
FOREIGN KEY (Sub_Code) REFERENCES SUBJECT(Sub_Code)
ON DELETE RESTRICT ON UPDATE CASCADE);

CREATE TABLE ASSIGNMENT (
Lec_ID CHAR(10) NOT NULL,
Sub_Code VARCHAR(9) NOT NULL,
PRIMARY KEY(Lec_ID, Sub_Code),
FOREIGN KEY(Lec_ID) REFERENCES LECTURER(Lec_ID)
ON DELETE RESTRICT ON UPDATE CASCADE,
FOREIGN KEY(Sub_Code) REFERENCES SUBJECT(Sub_Code)
ON DELETE RESTRICT ON UPDATE CASCADE);

CREATE TABLE REGISTRATION(
Std_ID CHAR(10) NOT NULL,
Sub_Code VARCHAR(9) NOT NULL,
PRIMARY KEY(Std_ID, Sub_Code),
FOREIGN KEY(Std_ID) REFERENCES STUDENT(Std_ID)
ON DELETE RESTRICT ON UPDATE CASCADE,
FOREIGN KEY(Sub_Code) REFERENCES SUBJECT(Sub_Code)
ON DELETE RESTRICT ON UPDATE CASCADE);

RETURNS INT

    DECLARE student_count INT;

    FROM REGISTRATION
    WHERE Sub_Code = subject_code;

    RETURN student_count;
END $$

e) For Procedure:
i.	This procedure, when called with a specific student ID as a parameter, will retrieve information about that student's exam schedule. It includes details such as the student's ID and name, the subject codes and names they are registered for, and the date, start time, end time, and venue information for the exams (if available). The use of LEFT JOIN ensures that even if a student is registered for subjects without exams or exams without venues, the information is still included in the result.

DELIMITER $$
CREATE PROCEDURE STUDExamScheduleReport(IN StudentID CHAR(10))
BEGIN
    SELECT
        STUDENT.Std_ID,
        STUDENT.Std_Name,
        REGISTRATION.Sub_Code,
        SUBJECT.Sub_Name,
        BOOKING.Book_Date,
        BOOKING.Book_Stime,
        BOOKING.Book_Etime,
        VENUE.Venue_Name
    FROM
        STUDENT
    JOIN
        REGISTRATION ON STUDENT.Std_ID = REGISTRATION.Std_ID
    JOIN
        SUBJECT ON REGISTRATION.Sub_Code = SUBJECT.Sub_Code
    LEFT JOIN
        BOOKING ON SUBJECT.Sub_Code = BOOKING.Sub_Code
    LEFT JOIN
        VENUE ON BOOKING.Venue_Code = VENUE.Venue_Code
    WHERE
        STUDENT.Std_ID = StudentID;
END$$
DELIMITER ;

call STUDExamScheduleReport(1211110647);

f)For trigger:
This trigger checks for booking conflicts when a new booking is inserted into the BOOKING table and raises an error if a conflict is found.

DELIMITER $$
CREATE TRIGGER BookingConflictChecking
BEFORE INSERT ON BOOKING
FOR EACH ROW
BEGIN
    DECLARE Conflict_Count INT;

    -- Check if there is a booking conflict for the same venue and time
    SELECT COUNT(*)
    INTO Conflict_Count
    FROM BOOKING
    WHERE Venue_Code = NEW.Venue_Code
      AND Book_Date = NEW.Book_Date
      AND ((NEW.Book_Stime < Book_Etime AND NEW.Book_Etime > Book_Stime)
           OR (NEW.Book_Etime > Book_Stime AND NEW.Book_Stime < Book_Etime));

    -- If conflict is found, raise error
    IF Conflict_Count > 0 THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Booking conflict found. Please choose different time slot or venue!!!';
    END IF;
END $$
DELIMITER ;
